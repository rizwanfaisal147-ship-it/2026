<script>
function go(n){
  document.getElementById('s'+n).classList.remove('active');
  document.getElementById('s'+(n+1)).classList.add('active');
}

function showMsg(){
  document.getElementById('msg').style.display='block';
}

/* ? FIXED COUNTDOWN (PAKISTAN TIME) */
const target = new Date("January 1, 2026 00:00:00 GMT+0500");

const timer = setInterval(()=>{
  const now = new Date();
  const diff = target - now;

  if(diff <= 0){
    clearInterval(timer);
    document.getElementById('countdown').innerText = "?? Happy New Year! ??";

    setTimeout(()=>{
      document.getElementById('s3').classList.remove('active');
      document.getElementById('s4').classList.add('active');
      document.getElementById('gift').classList.add('active');
    },1500);
    return;
  }

  const d = Math.floor(diff/(1000*60*60*24));
  const h = Math.floor((diff/(1000*60*60))%24);
  const m = Math.floor((diff/(1000*60))%60);
  const s = Math.floor((diff/1000)%60);

  document.getElementById('countdown').innerText =
    `${d}d ${h}h ${m}m ${s}s`;
},1000);

function openGift(){
  const gift = document.getElementById('gift');
  if(!gift.classList.contains('active')) return;

  gift.innerText = '??';
  document.getElementById('giftMsg').style.display='block';
}

  h.innerText='??';
  h.style.left=Math.random()*100+'vw';
  hearts.appendChild(h);
  setTimeout(()=>h.remove(),6000);
},500);
